\section{Introduction}

The Singapore University of Technology and Design (SUTD) has over 200 rooms split across 5 buildings.
To help people (especially visitors) find rooms,
each room is assigned an address of the form
$\texttt{x.y-z}$,
where $\texttt{x}$ is a building number,
$\texttt{y}$ is a floor number,
and $\texttt{z}$ is a room number.

However, resources that aim to map rooms to addresses remain few and far between.
Physical resources like campus signs exist, but require one to walk around, expending lots of energy and time.
Digital resources like the SUTD Assistive Maps (SAM) Telegram bot exist, but are often arduous to maintain and less robust to noise like typos (which can be especially pertinent for visitors who do not often type in English).

To fill this gap, we built SUTD 404, a digital resource that aims to maps rooms to addresses in a way that handles noise gracefully.
Our repository and web application can be found at
\begin{center}
\href{github.com/gregorylimeurhen/sutd404}{github.com/gregorylimeurhen/sutd404}
\end{center}
and
\begin{center}
\href{sutd404.vercel.app}{sutd404.vercel.app}
\end{center}
respectively.

Let $V$ be a vocabulary,
and $R\subseteq V^{*}$ be a room set,
and $A\subseteq V^{*}$ be an address set,
and $B\subseteq V$ be a boundary set,
and $C\in\N$ be a corruption count,
and $\alpha\in\R_{[0,1]}$ be a transposition rate,
and $\beta\in\R_{[0,1]}$ be a substitution rate,
and $\bm{\pi}=(\pi_\mathrm{train},\pi_\mathrm{val},\pi_\mathrm{test})\in\Delta^{2}$ be a training-validation-testing split,
and $I:V\rightharpoonup\Z^{2}$ be a keyboard-position map,
and $C\bm{\pi}\in\N^{3}$,
and each tie be broken uniform randomly.

\begin{algorithm}[H]
	\caption{$\textsc{Corrupt}(\bm{x})$}
	\begin{algorithmic}
		\Require $\bm{x}\in V^{*}$
		\State ${
			(T_{L},T_{M},T_{R})
			\gets
			(V^{*}B\cup\{()\},(V\setminus B)^{+},BV^{*}\cup\{()\})
		}$
		\State ${
			T
			\gets
			\{
				\bm{u}\bm{w}b\bm{v}\bm{z}
				\mid
				\bm{u}\in T_{L},\,
				\bm{v},\bm{w}\in T_{M},\,
				\bm{z}\in T_{R},\,
				b\in B,\,
				\bm{x}=\bm{u}\bm{v}b\bm{w}\bm{z}
			\}
			\setminus
			\{\bm{x}\}
		}$
		\If{$T\neq\varnothing$}
			${\bm{x}\sim\alpha\,\mathcal{U}(T)+(1-\alpha)\,\delta(\bm{x})}$
		\EndIf
		\For{$i\in[1..\abs{\bm{x}}]$}
			\State $S\gets\{k\in K\setminus \{x_{i}\}\mid I(k)-I(x_{i})\in[-1..1]\times[-2..2]\}$
			\If{$S\neq\varnothing$}
				${x_{i}\sim\beta\,\mathcal{U}(S)+(1-\beta)\,\delta(x_{i})}$
			\EndIf
		\EndFor
		\State \Return $\bm{x}$
	\end{algorithmic}
\end{algorithm}

\begin{algorithm}[H]
	\caption{$\textsc{Split}()$}
	\begin{algorithmic}
		\State $(D_\mathrm{train},D_\mathrm{val},D_\mathrm{test})\gets(\varnothing,\varnothing,\varnothing)$
		\For{$r\in R$}
			\State $P\gets\varnothing$
			\While{$\abs{P}<C$}
				\State $r'\sim\textsc{Corrupt}(r)$
				\If{$r'\neq r$}
					$P\gets P\cup\{(r',r)\}$
				\EndIf
			\EndWhile
			\State $P_\mathrm{train}\sim\mathcal{U}(\{Q\subseteq P\mid \abs{Q}=\pi_\mathrm{train}C\})$
			\State $P_\mathrm{val}\sim\mathcal{U}(\{Q\subseteq P\setminus P_\mathrm{train}\mid \abs{Q}=\pi_\mathrm{val}C\})$
			\State $P_\mathrm{test}\gets P\setminus(P_\mathrm{train}\cup P_\mathrm{val})$
			\State ${
				(D_\mathrm{train},D_\mathrm{val},D_\mathrm{test})
				\gets
				(
					D_\mathrm{train}\cup P_\mathrm{train},
					D_\mathrm{val}\cup P_\mathrm{val},
					D_\mathrm{test}\cup P_\mathrm{test}
				)
			}$
		\EndFor
		\State \Return $(D_\mathrm{train},D_\mathrm{val},D_\mathrm{test})$
	\end{algorithmic}
\end{algorithm}

Our corruption algorithm corrupts strings by transposing spans and substituting characters.
First, the algorithm samples whether to transpose spans by uniformly sampling from the space of span transpositions.
Then, for each character in the string, the algorithm samples whether to substitute the character by uniformly sampling from the space of character substitutions.
This space contains nearby characters collected by sliding a window of height $3$ and width $5$ centred at each position in $K$.
Finally, the algorithm returns the string.
We set $C=50$, and $\alpha=0.5$, and $\beta=0.05$, and $\bm{\pi}=(0.8,0.1,0.1)$ to generate plausible-looking typos tractably.

\begin{definition}[Denoiser]
Let $f:V^{*}\times R\to\R$ be a scorer.
\begin{align*}
	\mathscr{D}_{f}
	&:
	V^{*}\to R
	\\
	\mathscr{D}_{f}(x)
	&=
	\argmax_{r\in R}
	f(x,r)
\end{align*}
\end{definition}

\begin{definition}[Lookup]
$$
	\mathscr{L}:R\to A
$$
\end{definition}

% Edges map rooms to their addresses in the untouched `edges.tsv`
% Because a room like `foo` can have multiple addresses like `A, B`,
% we write a script `preprocess.py` to:
% 1. generate `train.tsv` (for D_train), `val.tsv` (for D_val), `test.tsv` (for D_test)
% by writing `corrupted	normal` pairs as rows
% 2. convert `edges.tsv` to `n2a.tsv` with each row like `foo	A, B`
% so look-up for `foo` yields `A, B`
% The data is thus fixed after preprocessing

Our denoiser is a Nanochat-parameterised, character-level transformer of depth 2 with decoding constrained to the trie induced by $R$. % cite karpathy/nanochat
We use character-level tokenisation as typos are often character-level, and room names are short enough for individual characters to be highly meaningful semantic units for our task.
We use trie-constrained decoding to efficiently avoid hallucination caused by unconstrained decoding.
We use a standard cross-entropy loss and AdamW optimiser.
A complete list of hyperparameters can be found in our GitHub repository.
Our lookup is a standard room-address table.

% Everything is clean, default, and standard with no other stuffs:
% - Training loss applied only to clean (target) portion
% - Adam on default settings
% - Batch size derivable as largest batch that fits in memory
% - Seed 1
% - Nothing else, no dropout or whatever

\begin{algorithm}[H]
	\caption{$\textsc{Infer}(f,x)$}
	\begin{algorithmic}
		\Require $f:V^{*}\times R\to\R$, $x\in V^{*}$
		\If{$x\not\in R$}
			$x\gets\mathscr{D}_{f}(x)$
		\EndIf
		\State $x\gets\mathscr{L}(x)$
		\State \Return $x$
	\end{algorithmic}
\end{algorithm}

Our inference algorithm receives a scorer and string as input.
Firstly, the algorithm denoises the string to a room using the scorer if the string is not a room.
Then, the algorithm looks up the address of the room.
Finally, the algorithm returns the address.

\section{Evaluation}

Damerau--Levenshtein distance (DL),
longest common substring length (LCS),
and longest common prefix (LCP) denoisers,

\section{Deployment}

\begin{algorithm}[H]
	\caption{$\textsc{Predict}(x)$}
	\begin{algorithmic}
		\Require $x\in V^{*}$
		\State $x\gets\textsc{Strip}(x)$
		% example: " LT2 " -> "LT2"
		\State $x\gets\textsc{Lowercase}(x)$
		% example: "LT2" -> "lt2"
		\State $x\gets\textsc{AlphanumericSplit}(x)$
		% example: "lt2" -> "lt 2"
		\State $x\gets\textsc{Alias}(x)$
		% example: "lt 2" -> "lecture theatre 2"
		\If{$x\in R$}
			\Return $(x,\mathscr{L}(x))$
		\EndIf
		\State ${ % do not refer to this, I will amend this later
			O
			\gets
			\bigcup_{
				f
				\in
				\{
					\textsc{L},
					\textsc{DL},
					\textsc{LCS},
					\textsc{CH},
					\textsc{Ours}
				\}
			}
			\arg\operatorname{top}_{r\in R}^{(2)}f(x,r)
		}$
		\State $O\gets\{(o,\mathscr{L}(o))\mid o\in O\}$
		\State $O\gets\textsc{LexicographicSort}(O)$
		\State \Return $O$
	\end{algorithmic}
\end{algorithm}

Our prediction algorithm receives a string as input.
Firstly, the algorithm normalises the string by stripping, lowercasing, separating letters from numbers, and replacing words with aliases, to more closely match strings at training.
Then, in parallel, the algorithm collects the top-2 rooms for the string under L, DL, LCS, CH, and an augmented version of our denoiser, as they have comparable latencies and are able to cover each others' weaknesses.
This augmented version of our denoiser uses beam search with beam width 2 to improve robustness at comparable latency. % Is this basically accurate?
Finally, the algorithm looks up the address of each room, returning the room-address pairs in lexicographic order for readability.

To improve the user experience of our application,
we made our application call the prediction algorithm on each new character entered and use prefix (key-value) caching,
to efficiently make the application feel more responsive.
